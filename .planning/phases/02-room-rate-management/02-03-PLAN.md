```python
    Raise 404 if no matching tier."""

async def get_seasonal_multiplier(db, room_type_id, target_date: date) -> Decimal:
    """Find active seasonal rate where start_date <= target_date <= end_date.
    Return multiplier if found, Decimal("1.00") if no seasonal rate applies."""

async def get_weekend_multiplier(db, room_type_id, target_date: date) -> Decimal:
    """Check if target_date.weekday() is in any active weekend surcharge's days list.
    Return multiplier if found, Decimal("1.00") if not."""

async def calculate_nightly_rate(db, room_type_id, target_date: date, occupancy: int) -> dict:
    """Calculate rate for one night with full breakdown.
    Returns {"base_amount": Decimal, "seasonal_multiplier": Decimal,
             "weekend_multiplier": Decimal, "final_amount": Decimal}
    final_amount = (base * seasonal * weekend).quantize(Decimal("0.01"))"""

async def calculate_stay_price(db, room_type_id, check_in: date, check_out: date, occupancy: int) -> dict:
    """Calculate total price for a date range.
    Iterates each night from check_in to check_out-1 (check_out is departure day).
    Returns {"currency": str, "nightly_rates": [NightlyRate...], "total": Decimal}"""
```

```python
async def lifespan(app: FastAPI):
    # MinIO bucket
    try:
        from app.core.storage import get_minio_client
        from app.services.storage import ensure_bucket
        from app.core.config import settings
        client = get_minio_client()
        ensure_bucket(client, settings.MINIO_BUCKET)

    # Seed data
    from app.core.config import settings
    if settings.SEED_ON_STARTUP:
        from app.core.database import async_session_factory
        from app.services.seed import seed_data
        async with async_session_factory() as session:

    yield
```
