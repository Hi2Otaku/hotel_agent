import asyncio
import uuid as uuid_mod
from fastapi import UploadFile
from minio import Minio

async def upload_photo(client: Minio, bucket: str, file: UploadFile) -> str:
    """Upload photo to MinIO, return object path. Runs sync client in executor."""
    ext = file.filename.rsplit(".", 1)[-1] if file.filename and "." in file.filename else "jpg"
    object_name = f"room-photos/{uuid_mod.uuid4()}.{ext}"
    loop = asyncio.get_event_loop()
    await loop.run_in_executor(
        None,
        lambda: client.put_object(
            bucket, object_name, file.file, length=-1,
            part_size=10 * 1024 * 1024, content_type=file.content_type or "image/jpeg",
        ),
    )
    return object_name

async def delete_photo(client: Minio, bucket: str, object_name: str) -> None:
    """Delete photo from MinIO."""
    loop = asyncio.get_event_loop()
    await loop.run_in_executor(None, lambda: client.remove_object(bucket, object_name))

def ensure_bucket(client: Minio, bucket: str) -> None:
    """Create bucket if it doesn't exist. Called on startup."""
    if not client.bucket_exists(bucket):
        client.make_bucket(bucket)
        # Set public read policy for demo
        import json
        policy = {
            "Version": "2012-10-17",
            "Statement": [{
                "Effect": "Allow",
                "Principal": {"AWS": ["*"]},
                "Action": ["s3:GetObject"],
                "Resource": [f"arn:aws:s3:::{bucket}/*"],
            }]
        }
        client.set_bucket_policy(bucket, json.dumps(policy))
```

from app.models.room import RoomStatus

ROLE_TRANSITIONS: dict[str, set[tuple[RoomStatus, RoomStatus]] | None] = {
    "front_desk": {
        (RoomStatus.AVAILABLE, RoomStatus.OCCUPIED),
        (RoomStatus.OCCUPIED, RoomStatus.AVAILABLE),
        (RoomStatus.RESERVED, RoomStatus.OCCUPIED),
    },
    "housekeeping": {
        (RoomStatus.CLEANING, RoomStatus.INSPECTED),
    },
    "manager": None,  # None = all transitions allowed
    "admin": None,
}

AUTO_TRANSITIONS = {
    "checkout": (RoomStatus.OCCUPIED, RoomStatus.CLEANING),
    "inspection_complete": (RoomStatus.INSPECTED, RoomStatus.AVAILABLE),
    "booking_assigned": (RoomStatus.AVAILABLE, RoomStatus.RESERVED),
}
```

Functions to implement:
- `async def create_room(db, data: RoomCreate) -> Room` -- Validate room_type_id exists (404 if not), check unique room_number (409 if duplicate).
- `async def get_room(db, room_id) -> Room | None`
- `async def get_rooms(db, skip, limit, floor: int | None, status: RoomStatus | None, room_type_id: uuid | None, active_only: bool) -> tuple[list[Room], int]` -- List with filters.
- `async def update_room(db, room_id, data: RoomUpdate) -> Room`
- `async def delete_room(db, room_id) -> None` -- Soft delete.
- `async def transition_status(db, room_id, new_status: RoomStatus, user_role: str, user_id: uuid, reason: str | None) -> Room`:
  1. Get room, raise 404 if not found
  2. If current status == new_status, return room (no-op)
  3. Look up `ROLE_TRANSITIONS[user_role]` -- if None (manager/admin), allow any transition. If set, check `(current_status, new_status)` is in the set. Raise 403 "Transition not allowed for your role" if not.
  4. Create `RoomStatusChange(room_id=room.id, from_status=room.status, to_status=new_status, changed_by=user_id, reason=reason)`
  5. Set `room.status = new_status`
  6. Commit and refresh
  7. Return room
- `async def get_status_board(db) -> dict` -- Query all active rooms joined with room_type. Group by floor. Return `{"floors": [...], "summary": {"available": N, "occupied": N, ...}}`.
- `async def get_status_history(db, room_id, limit=50) -> list[RoomStatusChange]` -- Return recent status changes for a room.

All endpoints under `router = APIRouter(prefix="/api/v1/rooms/types", tags=["room-types"])`.

Using `/api/v1/rooms/types` prefix so all room service routes fall under `/api/v1/rooms/*` which the gateway already maps.

Endpoints:
- `POST /api/v1/rooms/types` -- Create room type. Requires `require_manager_or_above`. Body: `RoomTypeCreate`. Returns `RoomTypeResponse`, status 201.
- `GET /api/v1/rooms/types` -- List room types. Requires `require_staff`. Query params: `skip: int = 0`, `limit: int = 50`, `active_only: bool = True`. Returns `RoomTypeListResponse`.
- `GET /api/v1/rooms/types/{room_type_id}` -- Get room type by ID. Requires `require_staff`. Returns `RoomTypeResponse`. Raise 404 if not found.
- `PATCH /api/v1/rooms/types/{room_type_id}` -- Update room type. Requires `require_manager_or_above`. Body: `RoomTypeUpdate`. Returns `RoomTypeResponse`.
- `DELETE /api/v1/rooms/types/{room_type_id}` -- Soft-delete room type. Requires `require_admin`. Returns 204 No Content.
- `POST /api/v1/rooms/types/{room_type_id}/photos` -- Upload photo. Requires `require_manager_or_above`. Accepts `UploadFile`. Uses `get_minio_client()` from storage, calls `upload_photo()`, then `add_photo_url()`. Returns updated `RoomTypeResponse`.
