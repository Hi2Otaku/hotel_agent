**templates/postgres-configmap.yaml** (init script for multi-DB setup):
Create a ConfigMap with a shell script `init-databases.sh` (NOT SQL, because we need to read env vars for passwords). The script:
```bash
#!/bin/bash
set -e

psql -v ON_ERROR_STOP=1 --username "$POSTGRES_USER" --dbname "$POSTGRES_DB" <<-EOSQL
    CREATE DATABASE rooms;
    CREATE DATABASE bookings;
    CREATE DATABASE chat;

    CREATE USER room_user WITH PASSWORD '$ROOM_DB_PASSWORD';
    CREATE USER booking_user WITH PASSWORD '$BOOKING_DB_PASSWORD';
    CREATE USER chat_user WITH PASSWORD '$CHAT_DB_PASSWORD';

    GRANT ALL PRIVILEGES ON DATABASE rooms TO room_user;
    GRANT ALL PRIVILEGES ON DATABASE bookings TO booking_user;
    GRANT ALL PRIVILEGES ON DATABASE chat TO chat_user;

    -- Connect to each database and grant schema privileges
    \c rooms
    GRANT ALL ON SCHEMA public TO room_user;
    \c bookings
    GRANT ALL ON SCHEMA public TO booking_user;
    \c chat
    GRANT ALL ON SCHEMA public TO chat_user;
EOSQL
```
Mount env vars ROOM_DB_PASSWORD, BOOKING_DB_PASSWORD, CHAT_DB_PASSWORD from the db-credentials secret on the Postgres pod.

**templates/postgres-statefulset.yaml:**
```yaml
apiVersion: apps/v1
kind: StatefulSet
metadata:
  name: postgres
  namespace: {{ include "hotelbook.namespace" . }}
  labels: {{- include "hotelbook.labels" . | nindent 4 }}
spec:
  serviceName: postgres
  replicas: 1
  selector:
    matchLabels:
      app: postgres
  template:
    metadata:
      labels:
        app: postgres
    spec:
      containers:
        - name: postgres
          image: {{ .Values.postgres.image }}
          ports:
            - containerPort: 5432
          env:
            - name: POSTGRES_DB
              value: {{ .Values.postgres.auth.database }}
            - name: POSTGRES_USER
              value: {{ .Values.postgres.auth.username }}
            - name: POSTGRES_PASSWORD
              valueFrom:
                secretKeyRef:
                  name: db-credentials
                  key: auth-db-password
            - name: ROOM_DB_PASSWORD
              valueFrom:
                secretKeyRef:
                  name: db-credentials
                  key: room-db-password
            - name: BOOKING_DB_PASSWORD
              valueFrom:
                secretKeyRef:
                  name: db-credentials
                  key: booking-db-password
            - name: CHAT_DB_PASSWORD
              valueFrom:
                secretKeyRef:
                  name: db-credentials
                  key: chat-db-password
            - name: PGDATA
              value: /var/lib/postgresql/data/pgdata
          volumeMounts:
            - name: postgres-data
              mountPath: /var/lib/postgresql/data
            - name: init-scripts
              mountPath: /docker-entrypoint-initdb.d
          resources:
            requests:
              memory: {{ .Values.postgres.resources.requests.memory }}
              cpu: {{ .Values.postgres.resources.requests.cpu }}
            limits:
              memory: {{ .Values.postgres.resources.limits.memory }}
              cpu: {{ .Values.postgres.resources.limits.cpu }}
          livenessProbe:
            exec:
              command: ["pg_isready", "-U", "{{ .Values.postgres.auth.username }}", "-d", "{{ .Values.postgres.auth.database }}"]
            initialDelaySeconds: 15
            periodSeconds: 10
          readinessProbe:
            exec:
              command: ["pg_isready", "-U", "{{ .Values.postgres.auth.username }}", "-d", "{{ .Values.postgres.auth.database }}"]
            initialDelaySeconds: 5
            periodSeconds: 5
      volumes:
        - name: init-scripts
          configMap:
            name: postgres-init
            defaultMode: 0755
  volumeClaimTemplates:
    - metadata:
        name: postgres-data
      spec:
        accessModes: ["ReadWriteOnce"]
        resources:
          requests:
            storage: {{ .Values.postgres.storage }}
```
