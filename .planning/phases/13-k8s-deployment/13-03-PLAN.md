**Replace the `build` job with `build-and-push`:**

```yaml
  build-and-push:
    runs-on: ubuntu-latest
    needs: e2e-tests
    if: github.ref == 'refs/heads/main' && github.event_name == 'push'
    permissions:
      packages: write
      contents: read
    strategy:
      matrix:
        service:
          - { name: auth, context: ".", dockerfile: services/auth/Dockerfile }
          - { name: room, context: ".", dockerfile: services/room/Dockerfile }
          - { name: booking, context: ".", dockerfile: services/booking/Dockerfile }
          - { name: chat, context: ".", dockerfile: services/chat/Dockerfile }
          - { name: mcp-server, context: ".", dockerfile: services/mcp-server/Dockerfile }
          - { name: gateway, context: ".", dockerfile: services/gateway/Dockerfile }
          - { name: frontend-guest, context: ./frontend, dockerfile: frontend/Dockerfile }
          - { name: frontend-staff, context: ./frontend-staff, dockerfile: frontend-staff/Dockerfile }
    steps:
      - uses: actions/checkout@v4

      - uses: docker/login-action@v3
        with:
          registry: ghcr.io
          username: ${{ github.actor }}
          password: ${{ secrets.GITHUB_TOKEN }}

      - uses: docker/build-push-action@v6
        with:
          context: ${{ matrix.service.context }}
          file: ${{ matrix.service.dockerfile }}
          push: true
          tags: |
            ghcr.io/${{ github.repository_owner }}/hotelbook-${{ matrix.service.name }}:${{ github.sha }}
            ghcr.io/${{ github.repository_owner }}/hotelbook-${{ matrix.service.name }}:latest
```

**Replace the `deploy` job:**

```yaml
  deploy:
    runs-on: ubuntu-latest
    needs: build-and-push
    if: github.ref == 'refs/heads/main' && github.event_name == 'push'
    steps:
      - name: Deploy to k3s
        uses: appleboy/ssh-action@v1
        env:
          IMAGE_TAG: ${{ github.sha }}
          REPO_OWNER: ${{ github.repository_owner }}
          AUTH_DB_PASSWORD: ${{ secrets.AUTH_DB_PASSWORD }}
          ROOM_DB_PASSWORD: ${{ secrets.ROOM_DB_PASSWORD }}
          BOOKING_DB_PASSWORD: ${{ secrets.BOOKING_DB_PASSWORD }}
          CHAT_DB_PASSWORD: ${{ secrets.CHAT_DB_PASSWORD }}
          RABBITMQ_PASS: ${{ secrets.RABBITMQ_PASS }}
          ADMIN_EMAIL: ${{ secrets.ADMIN_EMAIL }}
          ADMIN_PASSWORD: ${{ secrets.ADMIN_PASSWORD }}
          JWT_PRIVATE_KEY: ${{ secrets.JWT_PRIVATE_KEY }}
          JWT_PUBLIC_KEY: ${{ secrets.JWT_PUBLIC_KEY }}
          CHAT_LLM_PROVIDER: ${{ secrets.CHAT_LLM_PROVIDER }}
          CHAT_LLM_MODEL: ${{ secrets.CHAT_LLM_MODEL }}
          CHAT_LLM_API_KEY: ${{ secrets.CHAT_LLM_API_KEY }}
          MINIO_ROOT_USER: ${{ secrets.MINIO_ROOT_USER }}
          MINIO_ROOT_PASSWORD: ${{ secrets.MINIO_ROOT_PASSWORD }}
        with:
          host: ${{ secrets.EC2_HOST }}
          username: ${{ secrets.EC2_USER }}
          key: ${{ secrets.EC2_SSH_KEY }}
          envs: IMAGE_TAG,REPO_OWNER,AUTH_DB_PASSWORD,ROOM_DB_PASSWORD,BOOKING_DB_PASSWORD,CHAT_DB_PASSWORD,RABBITMQ_PASS,ADMIN_EMAIL,ADMIN_PASSWORD,JWT_PRIVATE_KEY,JWT_PUBLIC_KEY,CHAT_LLM_PROVIDER,CHAT_LLM_MODEL,CHAT_LLM_API_KEY,MINIO_ROOT_USER,MINIO_ROOT_PASSWORD
          script: |
            export KUBECONFIG=/etc/rancher/k3s/k3s.yaml
            cd /opt/hotelbook
            git pull origin main

            # Create namespace if not exists
            kubectl create namespace hotelbook --dry-run=client -o yaml | kubectl apply -f -

            # Create/update db-credentials secret
            kubectl create secret generic db-credentials \
              --namespace hotelbook \
              --from-literal=auth-db-password="$AUTH_DB_PASSWORD" \
              --from-literal=room-db-password="$ROOM_DB_PASSWORD" \
              --from-literal=booking-db-password="$BOOKING_DB_PASSWORD" \
              --from-literal=chat-db-password="$CHAT_DB_PASSWORD" \
              --from-literal=rabbitmq-password="$RABBITMQ_PASS" \
              --dry-run=client -o yaml | kubectl apply -f -

            # Create/update admin-credentials secret
            kubectl create secret generic admin-credentials \
              --namespace hotelbook \
              --from-literal=admin-email="$ADMIN_EMAIL" \
              --from-literal=admin-password="$ADMIN_PASSWORD" \
              --dry-run=client -o yaml | kubectl apply -f -

            # Create/update jwt-keys secret
            kubectl create secret generic jwt-keys \
              --namespace hotelbook \
              --from-literal=private-key="$JWT_PRIVATE_KEY" \
              --from-literal=public-key="$JWT_PUBLIC_KEY" \
              --dry-run=client -o yaml | kubectl apply -f -

            # Create/update chat-credentials secret
            kubectl create secret generic chat-credentials \
              --namespace hotelbook \
              --from-literal=llm-provider="$CHAT_LLM_PROVIDER" \
              --from-literal=llm-model="$CHAT_LLM_MODEL" \
              --from-literal=llm-api-key="$CHAT_LLM_API_KEY" \
              --dry-run=client -o yaml | kubectl apply -f -

            # Create/update minio-credentials secret
            kubectl create secret generic minio-credentials \
              --namespace hotelbook \
              --from-literal=root-user="$MINIO_ROOT_USER" \
              --from-literal=root-password="$MINIO_ROOT_PASSWORD" \
              --dry-run=client -o yaml | kubectl apply -f -

            # Create/update imagePullSecret for GHCR (if private repo)
            # Uses a PAT or GITHUB_TOKEN -- for public repos this is optional
            # kubectl create secret docker-registry ghcr-pull \
            #   --namespace hotelbook \
            #   --docker-server=ghcr.io \
            #   --docker-username="$REPO_OWNER" \
            #   --docker-password="$GHCR_PAT" \
            #   --dry-run=client -o yaml | kubectl apply -f -

            # Deploy with Helm
            helm upgrade --install hotelbook ./k8s/hotelbook \
              -n hotelbook \
              -f ./k8s/hotelbook/values-prod.yaml \
              --set global.imageTag="$IMAGE_TAG" \
              --set global.owner="$REPO_OWNER"

      - name: Verify deployment health
        uses: appleboy/ssh-action@v1
        with:
          host: ${{ secrets.EC2_HOST }}
          username: ${{ secrets.EC2_USER }}
          key: ${{ secrets.EC2_SSH_KEY }}
          script: |
            export KUBECONFIG=/etc/rancher/k3s/k3s.yaml

            echo "Waiting for rollout..."
            kubectl rollout status deployment/gateway -n hotelbook --timeout=120s || true

            echo "Pod status:"
            kubectl get pods -n hotelbook

            echo "Waiting for services to start..."
            sleep 15

            # Check health endpoint
            HTTP_STATUS=$(curl -s -o /dev/null -w "%{http_code}" http://localhost/health)
            if [ "$HTTP_STATUS" = "200" ]; then
              echo "Health check passed (HTTP $HTTP_STATUS)"
            else
              echo "Health check FAILED (HTTP $HTTP_STATUS)"
              kubectl logs -n hotelbook -l app=gateway --tail=50
              exit 1
            fi

            # Check guest frontend
            GUEST_STATUS=$(curl -s -o /dev/null -w "%{http_code}" http://localhost/)
            echo "Guest frontend: HTTP $GUEST_STATUS"

            # Check staff frontend
            STAFF_STATUS=$(curl -s -o /dev/null -w "%{http_code}" http://localhost/staff/)
            echo "Staff frontend: HTTP $STAFF_STATUS"

            echo "Deployment verified successfully"
```
