**templates/middleware.yaml:**
Create two Traefik Middleware CRDs:

```yaml
apiVersion: traefik.io/v1alpha1
kind: Middleware
metadata:
  name: strip-staff-prefix
  namespace: {{ include "hotelbook.namespace" . }}
spec:
  stripPrefix:
    prefixes:
      - /staff
---
apiVersion: traefik.io/v1alpha1
kind: Middleware
metadata:
  name: sse-headers
  namespace: {{ include "hotelbook.namespace" . }}
spec:
  headers:
    customResponseHeaders:
      X-Accel-Buffering: "no"
```

CRITICAL: Use API group `traefik.io/v1alpha1` (NOT `traefik.containo.us/v1alpha1`) -- k3s v1.32+ ships Traefik v3.

**templates/ingressroute.yaml:**
```yaml
apiVersion: traefik.io/v1alpha1
kind: IngressRoute
metadata:
  name: hotelbook
  namespace: {{ include "hotelbook.namespace" . }}
spec:
  entryPoints:
    - web
  routes:
    # SSE endpoint for chat streaming (highest priority)
    - match: PathPrefix(`/api/v1/chat/send`)
      kind: Rule
      services:
        - name: gateway
          port: 8000
          namespace: {{ include "hotelbook.namespace" . }}
      middlewares:
        - name: sse-headers
          namespace: {{ include "hotelbook.namespace" . }}

    # API routes (gateway)
    - match: PathPrefix(`/api`) || PathPrefix(`/docs`) || PathPrefix(`/openapi.json`) || PathPrefix(`/health`)
      kind: Rule
      services:
        - name: gateway
          port: 8000
          namespace: {{ include "hotelbook.namespace" . }}

    # Staff frontend (must come before guest catch-all)
    - match: PathPrefix(`/staff`)
      kind: Rule
      services:
        - name: frontend-staff
          port: 80
          namespace: {{ include "hotelbook.namespace" . }}
      middlewares:
        - name: strip-staff-prefix
          namespace: {{ include "hotelbook.namespace" . }}

    # Guest frontend (catch-all)
    - match: PathPrefix(`/`)
      kind: Rule
      services:
        - name: frontend-guest
          port: 80
          namespace: {{ include "hotelbook.namespace" . }}
```
