Docker path dependency resolution strategy:
- Use WORKDIR /build/services/{svc} so that ../../shared resolves to /build/shared/
- COPY shared/ /build/shared/
- COPY services/{svc}/pyproject.toml services/{svc}/uv.lock ./
- uv sync --frozen --no-dev --no-install-project (deps only layer)
- COPY services/{svc}/ . (source code layer)
- uv sync --frozen --no-dev (install project itself)
- ENV PATH="/build/services/{svc}/.venv/bin:$PATH"

    **Template for auth, room, booking, chat (services with entrypoint.sh):**
    ```dockerfile
    FROM python:3.12-slim

    # Install uv (pinned version)
    COPY --from=ghcr.io/astral-sh/uv:0.10.9 /uv /uvx /bin/

    # Configure uv for production
    ENV UV_COMPILE_BYTECODE=1 \
        UV_LINK_MODE=copy \
        UV_PYTHON_DOWNLOADS=never

    # Use WORKDIR that preserves relative path to shared/
    WORKDIR /build/services/{svc}

    # Layer 1: Copy shared package (changes rarely)
    COPY shared/ /build/shared/

    # Layer 2: Copy dependency manifests only (cached when deps unchanged)
    COPY services/{svc}/pyproject.toml services/{svc}/uv.lock ./

    # Layer 3: Install dependencies only (no project install yet)
    RUN --mount=type=cache,target=/root/.cache/uv \
        uv sync --frozen --no-dev --no-install-project

    # Layer 4: Copy service source code (changes frequently)
    COPY services/{svc}/ .

    # Layer 5: Install the project itself (picks up source)
    RUN --mount=type=cache,target=/root/.cache/uv \
        uv sync --frozen --no-dev

    # Activate the virtual environment
    ENV PATH="/build/services/{svc}/.venv/bin:$PATH"

    # Make entrypoint executable
    RUN chmod +x entrypoint.sh

    CMD ["./entrypoint.sh"]
    ```

    **Template for gateway (no entrypoint.sh, direct uvicorn CMD):**
    ```dockerfile
    FROM python:3.12-slim

    COPY --from=ghcr.io/astral-sh/uv:0.10.9 /uv /uvx /bin/

    ENV UV_COMPILE_BYTECODE=1 \
        UV_LINK_MODE=copy \
        UV_PYTHON_DOWNLOADS=never

    WORKDIR /build/services/gateway

    COPY shared/ /build/shared/
    COPY services/gateway/pyproject.toml services/gateway/uv.lock ./

    RUN --mount=type=cache,target=/root/.cache/uv \
        uv sync --frozen --no-dev --no-install-project

    COPY services/gateway/ .

    RUN --mount=type=cache,target=/root/.cache/uv \
        uv sync --frozen --no-dev

    ENV PATH="/build/services/gateway/.venv/bin:$PATH"

    CMD ["uvicorn", "app.main:app", "--host", "0.0.0.0", "--port", "8000"]
    ```

    Key differences from old Dockerfiles:
    - No `pip install` anywhere
    - No `ENV PYTHONPATH=/app` (uv handles this via venv)
    - WORKDIR is `/build/services/{svc}` not `/app` to preserve ../../shared path resolution
    - Two-phase `uv sync`: first without project (deps only, cached), then with project (after code copy)
    - `--mount=type=cache` for uv cache reuse across builds
    - `ENV PATH` adds .venv/bin for runtime

    Replace {svc} with auth, room, booking, gateway, chat respectively. Double-check that gateway has no entrypoint.sh and chat does have one.

    **Part B: Create root `Makefile`**

    Create `Makefile` at project root with these targets:
    ```makefile
    .PHONY: setup sync-all test lint test-auth test-room test-booking test-gateway test-chat clean

    SERVICES := auth room booking gateway chat

    ## Setup: Install uv and sync all services
    setup:
    	@command -v uv >/dev/null 2>&1 || { echo "Installing uv..."; curl -LsSf https://astral.sh/uv/install.sh | sh; }
    	@$(MAKE) sync-all

    ## Sync all service dependencies
    sync-all:
    	@for svc in $(SERVICES); do \
    		echo "Syncing $$svc..."; \
    		(cd services/$$svc && uv sync); \
    	done
    	@echo "All services synced."

    ## Run all backend tests
    test: test-auth test-room test-booking test-gateway test-chat

    ## Run auth service tests
    test-auth:
    	cd services/auth && uv run pytest tests/ -x --tb=short

    ## Run room service tests
    test-room:
    	cd services/room && uv run pytest tests/ -x --tb=short

    ## Run booking service tests
    test-booking:
    	cd services/booking && uv run pytest tests/ -x --tb=short

    ## Run gateway service tests
    test-gateway:
    	cd services/gateway && uv run pytest tests/ -x --tb=short

    ## Run chat service tests
    test-chat:
    	cd services/chat && uv run pytest tests/ -x --tb=short

    ## Lint all Python code
    lint:
    	uvx ruff check services/ shared/

    ## Lock all services (regenerate uv.lock files)
    lock-all:
    	@for svc in $(SERVICES); do \
    		echo "Locking $$svc..."; \
    		(cd services/$$svc && uv lock); \
    	done
    	@echo "All services locked."

    ## Clean virtual environments
    clean:
    	@for svc in $(SERVICES); do \
    		rm -rf services/$$svc/.venv; \
    	done
    	@echo "All .venv directories removed."
    ```
